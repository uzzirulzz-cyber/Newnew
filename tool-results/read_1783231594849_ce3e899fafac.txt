   110→// ===== Tooltip shared by recharts =====
   111→function ChartTooltip({ active, payload, label }: any) {
   112→  if (!active || !payload?.length) return null;
   113→  return (
   114→    <div className="rounded-lg border border-white/10 bg-black/80 px-3 py-2 text-xs backdrop-blur-xl">
   115→      {label != null && (
   116→        <p className="mb-1 font-medium text-white">{label}</p>
   117→      )}
   118→      {payload.map((p: any, i: number) => (
   119→        <p key={i} style={{ color: p.color }}>
   120→          {p.name}:{" "}
   121→          {typeof p.value === "number" ? p.value.toLocaleString() : p.value}
   122→        </p>
   123→      ))}
   124→    </div>
   125→  );
   126→}

   244→const BASE = "/api/v1";
   245→
   246→export interface ApiResult<T> {
   247→  success: boolean;
   248→  data?: T;
   249→  error?: { message: string; details?: unknown };
   250→}
   251→
   252→async function apiFetch<T>(
   253→  path: string,
   254→  options?: RequestInit
   255→): Promise<T> {
   256→  let res: Response;
   257→  try {
   258→    res = await fetch(`${BASE}${path}`, {
   259→      headers: {
   260→        "Content-Type": "application/json",
   261→        ...(options?.headers || {}),
   262→      },
   263→      ...options,
   264→    });
   265→  } catch (e) {
   266→    throw new Error("Network error. Please try again.");
   267→  }
   268→
   269→  let json: ApiResult<T> | null = null;
   270→  try {
   271→    json = (await res.json()) as ApiResult<T>;
   272→  } catch {
   273→    throw new Error(`Server returned status ${res.status}`);
   274→  }
   275→
   276→  if (!res.ok || !json?.success) {
   277→    const msg = json?.error?.message || `Request failed (${res.status})`;
   278→    throw new Error(msg);
   279→  }
   280→  if (!json.data) {
   281→    throw new Error("Empty response from server");
   282→  }
   283→  return json.data;
   284→}

   288→export interface ProductQuery {
   289→  search?: string;
   290→  category?: string;
   291→  type?: string;
   292→  sort?: string;
   293→  minPrice?: string;
   294→  maxPrice?: string;
   295→  featured?: string;
   296→  page?: number;
   297→  limit?: number;
   298→}
   299→
   300→export function buildProductQuery(q: ProductQuery): string {
   301→  const params = new URLSearchParams();
   302→  if (q.search) params.set("search", q.search);
   303→  if (q.category) params.set("category", q.category);
   304→  if (q.type) params.set("type", q.type);
   305→  if (q.sort) params.set("sort", q.sort);
   306→  if (q.minPrice) params.set("minPrice", q.minPrice);
   307→  if (q.maxPrice) params.set("maxPrice", q.maxPrice);
   308→  if (q.featured) params.set("featured", q.featured);
   309→  if (q.page) params.set("page", String(q.page));
   310→  if (q.limit) params.set("limit", String(q.limit));
   311→  const s = params.toString();
   312→  return s ? `?${s}` : "";
   313→}

   787→// ===== Currency system (PKR for Pakistan, USD elsewhere) =====
   788→
   789→export type Currency = "USD" | "PKR";
   790→
   791→// Conversion rate: 1 USD = 280 PKR. Only used when a product is priced in USD
   792→// and the customer wants to see PKR. NEVER used when the product is already
   793→// priced in PKR (DB products have currency="PKR" and price already in rupees).
   794→export const PKR_RATE = Number(process.env.NEXT_PUBLIC_PKR_RATE || 280);
   795→
   796→export const CURRENCY_META: Record<
   797→  Currency,
   798→  { code: Currency; symbol: string; locale: string; rate: number; label: string }
   799→> = {
   800→  USD: { code: "USD", symbol: "$", locale: "en-US", rate: 1, label: "USD" },
   801→  PKR: { code: "PKR", symbol: "Rs", locale: "en-PK", rate: PKR_RATE, label: "PKR" },
   802→};
   803→
   804→/**
   805→ * Format an amount that is ALREADY in the target currency.
   806→ *
   807→ * Use this for DB products where `price` is stored in PKR (currency="PKR")
   808→ * and we just want to render it as "Rs 2,232" — NO multiplication by 280.
   809→ *
   810→ * If the amount is in a DIFFERENT currency than the target, use
   811→ * `convertAndFormat()` below instead.
   812→ */
   813→export function formatInCurrency(
   814→  amount: number,
   815→  currency: Currency = "PKR",
   816→): string {
   817→  if (currency === "PKR") {
   818→    return `Rs ${Math.round(amount).toLocaleString("en-PK")}`;
   819→  }
   820→  return new Intl.NumberFormat("en-US", {
   821→    style: "currency",
   822→    currency: "USD",
   823→  }).format(amount);
   824→}
   825→
   826→/**
   827→ * Convert a USD amount to the target currency and format it.
   828→ *
   829→ * WARNING: This multiplies by 280 for PKR. Only use when the input is
   830→ * genuinely in USD. For DB products (already in PKR), use formatInCurrency().
   831→ *
   832→ * @deprecated Prefer formatInCurrency() + product.currency awareness.
   833→ */
   834→export function formatPrice(
   835→  usdAmount: number,
   836→  currency: Currency = "USD",
   837→): string {
   838→  const meta = CURRENCY_META[currency];
   839→  const converted = usdAmount * meta.rate;
   840→  if (currency === "PKR") {
   841→    return `Rs ${Math.round(converted).toLocaleString("en-PK")}`;
   842→  }
   843→  return new Intl.NumberFormat("en-US", {
   844→    style: "currency",
   845→    currency: "USD",
   846→  }).format(usdAmount);
   847→}
   848→
   849→/**
   850→ * Auto-detect the preferred currency from the browser timezone.
   851→ * Pakistan (Asia/Karachi) → PKR, everything else → USD.
   852→ */
   853→export function detectCurrency(): Currency {
   854→  if (typeof Intl === "undefined") return "USD";
   855→  try {
   856→    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone || "";
   857→    return tz === "Asia/Karachi" ? "PKR" : "USD";
   858→  } catch {
   859→    return "USD";
   860→  }
   861→}
   862→
   863→/**
   864→ * Display a product's price correctly.
   865→ *
   866→ * CRITICAL: DB products have `currency` field ("PKR" or "USD") and their
   867→ * `effectivePrice` is ALREADY in that currency. We must NOT multiply by 280.
   868→ *
   869→ * Lemon Squeezy products may have a `priceFormatted` string from the LS API
   870→ * (e.g. "PKR480/month") — use that directly if present.
   871→ *
   872→ * Only when a product is priced in USD AND the customer wants PKR do we
   873→ * convert (×280). Otherwise just format in the product's own currency.
   874→ */
   875→export function displayProductPrice(
   876→  product: {
   877→    priceFormatted?: string | null;
   878→    effectivePrice: number;
   879→    currency?: string | null;
   880→  },
   881→  displayCurrency: Currency = "PKR",
   882→): string {
   883→  // LS pre-formatted string — use as-is
   884→  if (product.priceFormatted) return product.priceFormatted;
   885→
   886→  const productCurrency: Currency =
   887→    product.currency === "USD" ? "USD" : "PKR";
   888→
   889→  // Product already in the customer's display currency — no conversion
   890→  if (productCurrency === displayCurrency) {
   891→    return formatInCurrency(product.effectivePrice, displayCurrency);
   892→  }
   893→
   894→  // Product is USD, customer wants PKR — convert (×280)
   895→  if (productCurrency === "USD" && displayCurrency === "PKR") {
   896→    return formatPrice(product.effectivePrice, "PKR");
   897→  }
   898→
   899→  // Product is PKR, customer wants USD — convert (÷280)
   900→  if (productCurrency === "PKR" && displayCurrency === "USD") {
   901→    return formatInCurrency(product.effectivePrice / PKR_RATE, "USD");
   902→  }
   903→
   904→  // Fallback: format in product's own currency
   905→  return formatInCurrency(product.effectivePrice, productCurrency);
   906→}

   244→const BASE = "/api/v1";
   245→
   246→export interface ApiResult<T> {
   247→  success: boolean;
   248→  data?: T;
   249→  error?: { message: string; details?: unknown };
   250→}
   251→
   252→async function apiFetch<T>(
   253→  path: string,
   254→  options?: RequestInit
   255→): Promise<T> {
   256→  let res: Response;
   257→  try {
   258→    res = await fetch(`${BASE}${path}`, {
   259→      headers: {
   260→        "Content-Type": "application/json",
   261→        ...(options?.headers || {}),
   262→      },
   263→      ...options,
   264→    });
   265→  } catch (e) {
   266→    throw new Error("Network error. Please try again.");
   267→  }
   268→
   269→  let json: ApiResult<T> | null = null;
   270→  try {
   271→    json = (await res.json()) as ApiResult<T>;
   272→  } catch {
   273→    throw new Error(`Server returned status ${res.status}`);
   274→  }
   275→
   276→  if (!res.ok || !json?.success) {
   277→    const msg = json?.error?.message || `Request failed (${res.status})`;
   278→    throw new Error(msg);
   279→  }
   280→  if (!json.data) {
   281→    throw new Error("Empty response from server");
   282→  }
   283→  return json.data;
   284→}

   288→export interface ProductQuery {
   289→  search?: string;
   290→  category?: string;
   291→  type?: string;
   292→  sort?: string;
   293→  minPrice?: string;
   294→  maxPrice?: string;
   295→  featured?: string;
   296→  page?: number;
   297→  limit?: number;

   299→
   300→export function buildProductQuery(q: ProductQuery): string {
   301→  const params = new URLSearchParams();
   302→  if (q.search) params.set("search", q.search);
   303→  if (q.category) params.set("category", q.category);
   304→  if (q.type) params.set("type", q.type);
   305→  if (q.sort) params.set("sort", q.sort);
   306→  if (q.minPrice) params.set("minPrice", q.minPrice);
   307→  if (q.maxPrice) params.set("maxPrice", q.maxPrice);
   308→  if (q.featured) params.set("featured", q.featured);
   309→  if (q.page) params.set("page", String(q.page));
   310→  if (q.limit) params.set("limit", String(q.limit));
   311→  const s = params.toString();
   312→  return s ? `?${s}` : "";
   313→}

   766→// ===== Currency system (PKR for Pakistan, USD elsewhere) =====
   767→
   768→export type Currency = "USD" | "PKR";
   769→
   770→// Conversion rate: 1 USD = 280 PKR. Adjust via PKR_RATE env if needed.
   771→export const PKR_RATE = Number(process.env.NEXT_PUBLIC_PKR_RATE || 280);
   772→
   773→export const CURRENCY_META: Record<
   774→  Currency,
   775→  { code: Currency; symbol: string; locale: string; rate: number; label: string }
   776→> = {
   777→  USD: { code: "USD", symbol: "$", locale: "en-US", rate: 1, label: "USD" },
   778→  PKR: { code: "PKR", symbol: "Rs", locale: "en-PK", rate: PKR_RATE, label: "PKR" },
   779→};
   780→
   781→/**
   782→ * Format a USD amount in the given display currency. PKR amounts are rounded
   783→ * to the nearest whole rupee (no decimals) per local convention.
   784→ */
   785→export function formatPrice(
   786→  usdAmount: number,
   787→  currency: Currency = "USD",
   788→): string {
   789→  const meta = CURRENCY_META[currency];
   790→  const converted = usdAmount * meta.rate;
   791→  if (currency === "PKR") {
   792→    return `Rs ${Math.round(converted).toLocaleString("en-PK")}`;
   793→  }
   794→  return new Intl.NumberFormat("en-US", {
   795→    style: "currency",
   796→    currency: "USD",
   797→  }).format(usdAmount);
   798→}
   799→
   800→/**
   801→ * Auto-detect the preferred currency from the browser timezone.
   802→ * Pakistan (Asia/Karachi) → PKR, everything else → USD.
   803→ */
   804→export function detectCurrency(): Currency {
   805→  if (typeof Intl === "undefined") return "USD";
   806→  try {
   807→    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone || "";
   808→    return tz === "Asia/Karachi" ? "PKR" : "USD";
   809→  } catch {
   810→    return "USD";
   811→  }
   812→}

   814→/**
   815→ * Display a product's price correctly.
   816→ *
   817→ * Lemon Squeezy products already have their price in the store's currency
   818→ * (e.g. PKR for the Pakistan store) and include a `priceFormatted` string
   819→ * from the LS API (e.g. "PKR480/month"). We MUST use that directly — never
   820→ * run it through formatPrice() which assumes the input is USD.
   821→ *
   822→ * For DB-seeded products (analytics only), use formatPrice() with the
   823→ * display currency.
   824→ */
   825→export function displayProductPrice(
   826→  product: { priceFormatted?: string | null; effectivePrice: number },
   827→  currency: Currency = "USD",
   828→): string {
   829→  if (product.priceFormatted) return product.priceFormatted;
   830→  return formatPrice(product.effectivePrice, currency);
   831→}

   246‚Üíexport interface ApiResult<T> {
   247‚Üí  success: boolean;
   248‚Üí  data?: T;
   249‚Üí  error?: { message: string; details?: unknown };
   250‚Üí}
   251‚Üí
   252‚Üíasync function apiFetch<T>(
   253‚Üí  path: string,
   254‚Üí  options?: RequestInit
   255‚Üí): Promise<T> {
   256‚Üí  let res: Response;
   257‚Üí  try {
   258‚Üí    res = await fetch(`${BASE}${path}`, {
   259‚Üí      headers: {
   260‚Üí        "Content-Type": "application/json",
   261‚Üí        ...(options?.headers || {}),
   262‚Üí      },
   263‚Üí      ...options,
   264‚Üí    });
   265‚Üí  } catch (e) {
   266‚Üí    throw new Error("Network error. Please try again.");
   267‚Üí  }
   268‚Üí
   269‚Üí  let json: ApiResult<T> | null = null;
   270‚Üí  try {
   271‚Üí    json = (await res.json()) as ApiResult<T>;
   272‚Üí  } catch {
   273‚Üí    throw new Error(`Server returned status ${res.status}`);
   274‚Üí  }
   275‚Üí
   276‚Üí  if (!res.ok || !json?.success) {
   277‚Üí    const msg = json?.error?.message || `Request failed (${res.status})`;
   278‚Üí    throw new Error(msg);
   279‚Üí  }
   280‚Üí  if (!json.data) {
   281‚Üí    throw new Error("Empty response from server");
   282‚Üí  }
   283‚Üí  return json.data;
   284‚Üí}

   728‚Üí// ===== Currency system (PKR for Pakistan, USD elsewhere) =====
   729‚Üí
   730‚Üíexport type Currency = "USD" | "PKR";
   731‚Üí
   732‚Üí// Conversion rate: 1 USD = 280 PKR. Adjust via PKR_RATE env if needed.
   733‚Üíexport const PKR_RATE = Number(process.env.NEXT_PUBLIC_PKR_RATE || 280);
   734‚Üí
   735‚Üíexport const CURRENCY_META: Record<
   736‚Üí  Currency,
   737‚Üí  { code: Currency; symbol: string; locale: string; rate: number; label: string }
   738‚Üí> = {
   739‚Üí  USD: { code: "USD", symbol: "$", locale: "en-US", rate: 1, label: "USD" },
   740‚Üí  PKR: { code: "PKR", symbol: "Rs", locale: "en-PK", rate: PKR_RATE, label: "PKR" },
   741‚Üí};
   742‚Üí
   743‚Üí/**
   744‚Üí * Format a USD amount in the given display currency. PKR amounts are rounded
   745‚Üí * to the nearest whole rupee (no decimals) per local convention.
   746‚Üí */
   747‚Üíexport function formatPrice(
   748‚Üí  usdAmount: number,
   749‚Üí  currency: Currency = "USD",
   750‚Üí): string {
   751‚Üí  const meta = CURRENCY_META[currency];
   752‚Üí  const converted = usdAmount * meta.rate;
   753‚Üí  if (currency === "PKR") {
   754‚Üí    return `Rs ${Math.round(converted).toLocaleString("en-PK")}`;
   755‚Üí  }
   756‚Üí  return new Intl.NumberFormat("en-US", {
   757‚Üí    style: "currency",
   758‚Üí    currency: "USD",
   759‚Üí  }).format(usdAmount);
   760‚Üí}

   762‚Üí/**
   763‚Üí * Auto-detect the preferred currency from the browser timezone.
   764‚Üí * Pakistan (Asia/Karachi) ‚Üí PKR, everything else ‚Üí USD.
   765‚Üí */
   766‚Üíexport function detectCurrency(): Currency {
   767‚Üí  if (typeof Intl === "undefined") return "USD";
   768‚Üí  try {
   769‚Üí    const tz = Intl.DateTimeFormat().resolvedOptions().timeZone || "";
   770‚Üí    return tz === "Asia/Karachi" ? "PKR" : "USD";
   771‚Üí  } catch {
   772‚Üí    return "USD";
   773‚Üí  }
   774‚Üí}

   776‚Üí/**
   777‚Üí * Display a product's price correctly.
   778‚Üí *
   779‚Üí * Lemon Squeezy products already have their price in the store's currency
   780‚Üí * (e.g. PKR for the Pakistan store) and include a `priceFormatted` string

   782‚Üí * run it through formatPrice() which assumes the input is USD.
   783‚Üí *
   784‚Üí * For DB-seeded products (analytics only), use formatPrice() with the

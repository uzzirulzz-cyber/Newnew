   159→function ChartTooltip({ active, payload, label }: any) {
   160→  if (!active || !payload?.length) return null;
   161→  return (
   162→    <div className="rounded-lg border border-white/10 bg-black/80 px-3 py-2 text-xs backdrop-blur-xl">
   163→      <p className="mb-1 font-medium text-white">{label}</p>
   164→      {payload.map((p: any, i: number) => (
   165→        <p key={i} style={{ color: p.color }}>
   166→          {p.name}: {p.value.toLocaleString()}
   167→        </p>
   168→      ))}
   169→    </div>
   170→  );
   171→}

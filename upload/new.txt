export default function PublicLayout({ children }: { children: React.ReactNode }) {
  const location = useLocation();
  const [mobileOpen, setMobileOpen] = useState(false);

  const headerRaw = useQuery(api.settings.get, { key: "header" });
  const footerRaw = useQuery(api.settings.get, { key: "footer" });

  const header = (headerRaw ?? {}) as HeaderSettings;
  const footer = (footerRaw ?? {}) as FooterSettings;

  const siteName = header.siteName ?? "SiteBuilder";
  const navItems: NavItem[] = header.navItems?.length ? header.navItems : defaultNav;
  const footerLinks: NavItem[] = footer.links ?? defaultNav;

            </div>
            {siteName}
          </Link>

          {/* Desktop nav */}
          <nav className="hidden md:flex items-center gap-1 ml-6">
            {navItems.map((item) => (
              <Link
                key={item.href}
                to={item.href}
                className={cn(

                  location.pathname === item.href

                    : "text-muted-foreground hover:text-foreground hover:bg-muted"
                )}
              >
                {item.label}
              </Link>
            ))}
          </nav>

          <div className="flex-1" />

          {/* Mobile menu toggle */}
          <button
            className="md:hidden text-muted-foreground hover:text-foreground transition-colors"

          >
            {mobileOpen ? <X className="w-5 h-5" /> : <Menu className="w-5 h-5" />}

                to={item.href}

export const list = query({
  args: {},
  handler: async (ctx) => {
    return await ctx.db.query("jobListings").withIndex("by_order").order("asc").collect();
  },
});

export const upsert = mutation({
  args: {
    id: v.optional(v.id("jobListings")),
    title: v.string(),
    department: v.optional(v.string()),
    location: v.optional(v.string()),
    type: v.optional(v.string()),
    description: v.string(),
    published: v.boolean(),
    order: v.number(),
  },
  handler: async (ctx, args) => {
    if (args.id) {
      const { id, ...data } = args;
      await ctx.db.patch(id, data);
      return id;
    }
    const { id: _id, ...data } = args;
    return await ctx.db.insert("jobListings", data);
  },
});

export const remove = mutation({
  args: { id: v.id("jobListings") },
  handler: async (ctx, args) => {
    await ctx.db.delete(args.id);
  },
});import { mutation, query } from "./_generated/server";
import { v } from "convex/values";

export const get = query({
  args: { slug: v.string() },
  handler: async (ctx, args) => {
    return await ctx.db
      .query("staticPages")
      .withIndex("by_slug", (q) => q.eq("slug", args.slug))
      .unique();
  },
});

export const upsert = mutation({
  args: {
    slug: v.string(),
    title: v.string(),
    content: v.string(),
  },
  handler: async (ctx, args) => {
    const existing = await ctx.db
      .query("staticPages")
      .withIndex("by_slug", (q) => q.eq("slug", args.slug))
      .unique();
    const updatedAt = new Date().toISOString();
    if (existing) {
      await ctx.db.patch(existing._id, { title: args.title, content: args.content, updatedAt });
    } else {
      await ctx.db.insert("staticPages", { ...args, updatedAt });
    }
  },
});import { mutation, query } from "./_generated/server";

export const get = query({
  args: {},
  handler: async (ctx) => {
    const results = await ctx.db.query("contactSettings").take(1);

  },
});

export const upsert = mutation({
  args: {
    email: v.optional(v.string()),
    phone: v.optional(v.string()),
    address: v.optional(v.string()),
    mapEmbed: v.optional(v.string()),
    formEnabled: v.boolean(),
  },
  handler: async (ctx, args) => {
    const results = await ctx.db.query("contactSettings").take(1);
    if (results[0]) {
      await ctx.db.patch(results[0]._id, args);
    } else {
      await ctx.db.insert("contactSettings", args);
    }
  },
